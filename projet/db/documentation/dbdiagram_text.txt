Enum jour_semaine{
  lundi
  mardi
  mercredi
  jeudi
  vendredi
  samedi
  dimanche
}

Table ressource {
  label varchar [pk]
  description text
}

Table jour_horaire {
  id integer [pk, increment]
  jour jour_semaine [not null]
  debut time [not null] 
  fin time [not null]
  temps_reservation time [not null]
  label varchar
}



Table temp_reservation {
  label_ressource varchar [not null, pk]
  heure time [not null, pk]
  date_reserv date [not null, pk]
  timestamp_reserv timestamp [not null]
}

Table reservation {
  id integer [pk, increment]
  nom varchar [not null]
  prenom varchar [not null]
  numero_tel varchar [not null]
  date_reservation timestamp [not null]
}

Table reservations_client_resource {
  reservation integer [not null]
  resource varchar [pk, not null]
  date_reservation date [pk, not null]
  heure time [pk, not null]
}

Table options_resource {
  label_resource varchar [not null, pk]
  label_option varchar [not null, pk]
  description text
}

Table options_resource_choix {
  id integer [pk, increment]
  label_option varchar [not null]
  choix varchar [not null]
}
